package main;

public class Main01 {

	public static void main(String[] args) {
		int contador = 1;
		
		while(contador <= 10) {
			System.out.println(contador);
			contador = contador + 1;
			//contador++;
		}
		
		contador = 1;
		do {
			System.out.println(contador);
			contador = contador + 1;
		} while(contador <= 10);
	}

}
